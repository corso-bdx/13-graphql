query {
  libro(
    where: {
      editore: { nome: { neq: "Einaudi" } }
      and: {
        autore: { inAttivita: { eq: true } }
        and: { costo: { gte: 10 }, and: { costo: { lte: 100 } } }
      }
    }
    order: { prezzo: ASC, titolo: DESC }
  ) {
    codice
    titolo
    prezzo
    autore {
      nome
      cognome
      inAttivita
    }
    casaEditrice {
      nome
    }
  }
}
