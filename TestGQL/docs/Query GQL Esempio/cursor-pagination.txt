query {
  a: libro (first: 1 after: "MQ=="
    where: {
      editore: { nome: { neq: "Einaudi" } }
      and: {
        autore: { inAttivita: { eq: true } }
        and: { costo: { gte: 10 }, and: { costo: { lte: 100 } } }
      }
    }
    order: { costo: ASC, titolo: DESC }
  ) {
		edges {
			node {
				codice
				titolo
				prezzo
				autore {
					nome
					cognome
					inAttivita
				}
				casaEditrice {
					nome
				}
			}	
			cursor
		}
		pageInfo {
			hasNextPage
			hasPreviousPage
			startCursor
			endCursor
		}
		
  }
}
